<script>
// CLOCK
setInterval(()=>clock.innerText="üïí "+new Date().toLocaleTimeString(),1000);

// üéµ MUSIC VISUALIZER
let audio,ctx,analyser,data;
const canvas=document.getElementById("viz");
const ctx2=canvas.getContext("2d");

fileInput.onchange=e=>{
 audio=new Audio(URL.createObjectURL(e.target.files[0]));
 ctx=new AudioContext();
 let src=ctx.createMediaElementSource(audio);
 analyser=ctx.createAnalyser();
 src.connect(analyser);
 analyser.connect(ctx.destination);
 data=new Uint8Array(analyser.frequencyBinCount);
};

function playMusic(){ if(!audio) return; audio.play(); visualize(); }
function pauseMusic(){ if(audio) audio.pause(); }

function visualize(){
 analyser.getByteFrequencyData(data);
 ctx2.clearRect(0,0,canvas.width,canvas.height);
 let barWidth=canvas.width/data.length;
 for(let i=0;i<data.length;i+=5){
   let h=data[i]/2;
   ctx2.fillStyle="rgb(0,"+(h+150)+",255)";
   ctx2.fillRect(i*barWidth,canvas.height-h,barWidth-2,h);
 }
 requestAnimationFrame(visualize);
}

// üéÆ 3D ENGINE
let scene=new THREE.Scene();
let camera=new THREE.PerspectiveCamera(75,400/300,0.1,1000);
let renderer=new THREE.WebGLRenderer();
renderer.setSize(400,300);
document.getElementById("game3d").appendChild(renderer.domElement);

let cube=new THREE.Mesh(
 new THREE.BoxGeometry(),
 new THREE.MeshBasicMaterial({color:0x00bfff})
);
scene.add(cube);
camera.position.z=3;

function animate(){
 requestAnimationFrame(animate);
 cube.rotation.x+=0.01;
 cube.rotation.y+=0.01;
 renderer.render(scene,camera);
}
animate();

// SPORTS
function searchSport(){
 let s=sportBox.value.toLowerCase();
 sportOut.innerText = s.includes("voetbal")?"‚öΩ Voetbal info":
 s.includes("basket")?"üèÄ Basket info":
 s.includes("handbal")?"ü§æ Handbal info":"Geen data";
}

async function getScore(team){
 let res=await fetch("https://www.thesportsdb.com/api/v1/json/3/searchteams.php?t="+team);
 let data=await res.json();
 if(data.teams) sportOut.innerText="League: "+data.teams[0].strLeague;
}
